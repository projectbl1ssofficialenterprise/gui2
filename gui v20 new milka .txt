        local success, response = pcall(function()
            return HttpService:JSONDecode(game:HttpGet(url))
        end)
        if success and response and response.data and #response.data > 0 then
            local servers = response.data
            local randomServer = servers[math.random(1, #servers)]

                TeleportService:TeleportToPlaceInstance(game.PlaceId, randomServer.id)
            end
        end
    end)
end)

local fpsBtn = makeUtilBtn("FPS Counter: OFF", leftUtility)
local fpsConnection
fpsBtn.MouseButton1Click:Connect(function()
    if fpsConnection then
        fpsConnection:Disconnect()
        fpsConnection = nil
        fpsBtn.Text = "FPS Counter: OFF"
        fpsBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        fpsBtn.TextColor3 = Color3.new(1, 1, 1)
        addLog("FPS Counter disabled")
        return
    end

    local frames = 0
    local lastTime = tick()
    fpsConnection = game:GetService("RunService").Heartbeat:Connect(function()
        frames += 1
        if tick() - lastTime >= 1 then
            local fps = math.floor(frames / (tick() - lastTime) + 0.5)
            fpsBtn.Text = "FPS: " .. fps
            if fps >= 60 then
                fpsBtn.TextColor3 = Color3.fromRGB(0, 255, 0)
            elseif fps >= 30 then
                fpsBtn.TextColor3 = Color3.fromRGB(255, 255, 0)
            else
                fpsBtn.TextColor3 = Color3.fromRGB(255, 0, 0)
            end
            frames = 0
            lastTime = tick()
        end
    end)
    fpsBtn.BackgroundColor3 = Color3.fromRGB(115, 12, 194)
    addLog("FPS Counter enabled")
end)
