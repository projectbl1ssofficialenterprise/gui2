local function updateLineNumbers()
    local text = code.Text or ""
    local lines = 1
    for _ in text:gmatch("\n") do lines = lines + 1 end
    local numText = ""
    for i = 1, lines do
        numText = numText .. tostring(i) .. "\n"
    end
    lineNumbers.Text = numText
end
